        const sheetUrl = "https://docs.google.com/spreadsheets/d/1NToggDh4KKhGiFNc7mgqF_V79FPDJT2qbcQZ6QAWmEw/export?format=csv&gid=0";


        async function fetchSeatBalance() {

            try {

                const response = await fetch(sheetUrl);

                const text = await response.text();

                const rows = text.split("\n");

                const seatBalance = rows[0].split(",")[0]; // ดึงข้อมูลจาก A1

                document.getElementById("seatBalance").innerText = seatBalance || "ไม่มีข้อมูล"; // แสดงข้อความถ้าไม่มีข้อมูล

                document.querySelector('.progress-2').style.display = 'none'; // ซ่อนแอนิเมชันเมื่อโหลดเสร็จ

            } catch (error) {

                console.error("Error fetching data:", error);

                document.getElementById("seatBalance").innerText = "Error loading data.";

                document.querySelector('.progress-2').style.display = 'none'; // ซ่อนแอนิเมชันเมื่อเกิดข้อผิดพลาด

            }

        }


        // Redirect to Line Link after 4 seconds

        function redirectToLine() {

        }


        // Execute Functions

        fetchSeatBalance();
